Write a C++ program to remove html tags from a string.

            #include < iostream >
            #include < string >
            
            std::string removeHtmlTags(const std::string& htmlString) {
                std::string result;
                bool insideTag = false;
            
                for (char c : htmlString) {
                    if (c == '<') {
                        insideTag = true;
                    } else if (c == '>') {
                        insideTag = false;
                    } else if (!insideTag) {
                        result += c;
                    }
                }
            
                return result;
            }
            
            int main() {
                std::string htmlString;
            
                // Input
                std::cout << "Enter an HTML string: ";
                std::getline(std::cin, htmlString);
            
                // Remove HTML tags
                std::string cleanedString = removeHtmlTags(htmlString);
            
                // Output
                std::cout << "String after removing HTML tags: " << cleanedString << std::endl;
            
                return 0;
            }
