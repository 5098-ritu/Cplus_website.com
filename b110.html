Write a C++ program to round up the result of integer division:


            #include < iostream >

            int main() {
                int dividend, divisor;
            
                // Input
                std::cout << "Enter the dividend: ";
                std::cin >> dividend;
                std::cout << "Enter the divisor: ";
                std::cin >> divisor;
            
                // Calculate quotient
                int quotient = dividend / divisor;
            
                // Calculate remainder
                int remainder = dividend % divisor;
            
                // If remainder is non-zero, round up
                if (remainder != 0) {
                    quotient++;
                }
            
                // Output
                std::cout << "Quotient after rounding up: " << quotient << std::endl;
            
                return 0;
            }

Output:


            Enter the dividend: 10
            Enter the divisor: 3
            Quotient after rounding up: 4
