Write a C++ program to convert Roman number to an integer number.

            #include < iostream >
            #include < unordered_map >
            
            int romanToInt(std::string s) {
                std::unordered_map romanMap = {
                    {'I', 1},
                    {'V', 5},
                    {'X', 10},
                    {'L', 50},
                    {'C', 100},
                    {'D', 500},
                    {'M', 1000}
                };
            
                int result = 0;
                int prevValue = 0;
            
                for (int i = s.length() - 1; i >= 0; --i) {
                    int value = romanMap[s[i]];
                    if (value < prevValue) {
                        result -= value;
                    } else {
                        result += value;
                    }
                    prevValue = value;
                }
            
                return result;
            }
            
            int main() {
                std::string romanNumeral;
                
                // Input
                std::cout << "Enter a Roman numeral: ";
                std::cin >> romanNumeral;
                
                // Convert to integer
                int integerNumber = romanToInt(romanNumeral);
                
                // Output
                std::cout << "Integer representation: " << integerNumber << std::endl;
            
                return 0;
            }

Output:
