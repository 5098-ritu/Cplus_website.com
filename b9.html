9. Write a C++ program to check whether C++ is installed on your computer or not:

            #include < iostream >
            #include < cstdlib >
            
            int main() {
                // Attempt to compile a simple C++ program
                int result = system("tcc -c -");
                
                if (result == 0) {
                    std::cout << "Turbo C++ is installed on your computer." << std::endl;
                } else {
                    std::cout << "Turbo C++ is not installed on your computer." << std::endl;
                }
            
                return 0;
            }

Output:

            Turbo C++ is not installed on your computer.
